> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 为当前 Drone plugin 增加基于 `codex` / `claude` CLI 的 AI 评审能力，支持 `pr` 输出结构化 JSON、`full` 输出 Markdown。

**Architecture:** 保持单一插件入口，对外通过 `PLUGIN_MODE` 和 `PLUGIN_PROVIDER` 切换行为；对内拆成配置解析、Git 上下文、Prompt 构造、Provider 适配、结果渲染五个模块。`pr` 模式依赖 Drone PR 环境变量与本地 Git diff，`full` 模式直接在已 clone 的仓库工作区内调用 CLI 做仓库级审查。

**Tech Stack:** Go 1.26、`urfave/cli`、`logrus`、本地 Git、`codex` CLI、`claude` CLI、Go `testing`

```go
package plugin

import (
	"fmt"
	"os"
	"path/filepath"
	"strings"
)

type Mode string

const (
	ModePR   Mode = "pr"
	ModeFull Mode = "full"
)

type ProviderName string

const (
	ProviderCodex  ProviderName = "codex"
	ProviderClaude ProviderName = "claude"
)

type Config struct {
	Mode        Mode
	Provider    ProviderName
	Output      string
	WorkDir     string
	PromptFile  string
	CLIBin      string
	Model       string
	ExtraArgs   []string
	Include     []string
	Exclude     []string
	Debug       bool
}

func ParseConfig() (Config, error) {
	cfg := Config{
		Mode:       Mode(strings.TrimSpace(os.Getenv("PLUGIN_MODE"))),
		Provider:   ProviderName(strings.TrimSpace(os.Getenv("PLUGIN_PROVIDER"))),
		Output:     strings.TrimSpace(os.Getenv("PLUGIN_OUTPUT")),
		PromptFile: strings.TrimSpace(os.Getenv("PLUGIN_PROMPT_FILE")),
		CLIBin:     strings.TrimSpace(os.Getenv("PLUGIN_CLI_BIN")),
		Model:      strings.TrimSpace(os.Getenv("PLUGIN_MODEL")),
		Debug:      strings.TrimSpace(os.Getenv("PLUGIN_DEBUG")) != "",
	}

	wd := strings.TrimSpace(os.Getenv("PLUGIN_WORKDIR"))
	if wd == "" {
		current, err := os.Getwd()
		if err != nil {
			return Config{}, fmt.Errorf("get workdir: %w", err)
		}
		wd = current
	}
	cfg.WorkDir = filepath.Clean(wd)
	cfg.ExtraArgs = splitCSVLike(os.Getenv("PLUGIN_EXTRA_ARGS"))
	cfg.Include = splitCSVLike(os.Getenv("PLUGIN_INCLUDE"))
	cfg.Exclude = splitCSVLike(os.Getenv("PLUGIN_EXCLUDE"))

	if cfg.Mode != ModePR && cfg.Mode != ModeFull {
		return Config{}, fmt.Errorf("PLUGIN_MODE must be one of: pr, full")
	}
	if cfg.Provider != ProviderCodex && cfg.Provider != ProviderClaude {
		return Config{}, fmt.Errorf("PLUGIN_PROVIDER must be one of: codex, claude")
	}
	if cfg.Output == "" {
		return Config{}, fmt.Errorf("PLUGIN_OUTPUT is required")
	}
	if cfg.CLIBin == "" {
		cfg.CLIBin = string(cfg.Provider)
	}
	return cfg, nil
}

func splitCSVLike(raw string) []string {
	fields := strings.FieldsFunc(raw, func(r rune) bool {
		return r == ',' || r == '\n'
	})
	var out []string
	for _, field := range fields {
		value := strings.TrimSpace(field)
		if value != "" {
			out = append(out, value)
		}
	}
	return out
}
```

```go
package plugin

import (
	"encoding/json"
	"fmt"
	"strings"
)

type PRReview struct {
	Verdict  string      `json:"verdict"`
	Summary  string      `json:"summary"`
	Findings []PRFinding `json:"findings"`
}

type PRFinding struct {
	Path      string `json:"path"`
	LineStart int    `json:"line_start"`
	LineEnd   int    `json:"line_end"`
	Severity  string `json:"severity"`
	Title     string `json:"title"`
	Body      string `json:"body"`
}

func ParsePRReview(raw []byte) (PRReview, error) {
	var review PRReview
	if err := json.Unmarshal(raw, &review); err != nil {
		return PRReview{}, fmt.Errorf("decode pr review json: %w", err)
	}
	if review.Verdict != "pass" && review.Verdict != "needs_attention" {
		return PRReview{}, fmt.Errorf("invalid verdict %q", review.Verdict)
	}
	if strings.TrimSpace(review.Summary) == "" {
		return PRReview{}, fmt.Errorf("summary is required")
	}
	for index, finding := range review.Findings {
		if strings.TrimSpace(finding.Path) == "" {
			return PRReview{}, fmt.Errorf("findings[%d].path is required", index)
		}
		if finding.LineStart <= 0 || finding.LineEnd <= 0 {
			return PRReview{}, fmt.Errorf("findings[%d] line_start and line_end must be positive", index)
		}
		if finding.LineEnd < finding.LineStart {
			return PRReview{}, fmt.Errorf("findings[%d] line_end must be >= line_start", index)
		}
		switch finding.Severity {
		case "low", "medium", "high":
		default:
			return PRReview{}, fmt.Errorf("findings[%d] invalid severity %q", index, finding.Severity)
		}
		if strings.TrimSpace(finding.Title) == "" {
			return PRReview{}, fmt.Errorf("findings[%d].title is required", index)
		}
		if strings.TrimSpace(finding.Body) == "" {
			return PRReview{}, fmt.Errorf("findings[%d].body is required", index)
		}
	}
	return review, nil
}
```

```go
package plugin

import (
	"bytes"
	"fmt"
	"os"
	"os/exec"
	"strings"
)

type PRContext struct {
	Number       string
	SourceBranch string
	TargetBranch string
}

func LoadPRContext() (PRContext, error) {
	ctx := PRContext{
		Number:       strings.TrimSpace(os.Getenv("DRONE_PULL_REQUEST")),
		SourceBranch: strings.TrimSpace(os.Getenv("DRONE_SOURCE_BRANCH")),
		TargetBranch: strings.TrimSpace(os.Getenv("DRONE_TARGET_BRANCH")),
	}
	if ctx.Number == "" {
		return PRContext{}, fmt.Errorf("DRONE_PULL_REQUEST is required in pr mode")
	}
	if ctx.SourceBranch == "" {
		return PRContext{}, fmt.Errorf("DRONE_SOURCE_BRANCH is required in pr mode")
	}
	if ctx.TargetBranch == "" {
		return PRContext{}, fmt.Errorf("DRONE_TARGET_BRANCH is required in pr mode")
	}
	return ctx, nil
}

func BuildPRDiff(workDir, targetBranch string) (string, error) {
	if err := runGit(workDir, "fetch", "origin", targetBranch); err != nil {
		return "", fmt.Errorf("fetch target branch: %w", err)
	}
	diff, err := runGitOutput(workDir, "diff", "--merge-base", "origin/"+targetBranch, "HEAD")
	if err != nil {
		return "", fmt.Errorf("build diff: %w", err)
	}
	if strings.TrimSpace(diff) == "" {
		return "", fmt.Errorf("git diff is empty")
	}
	return diff, nil
}

func runGit(dir string, args ...string) error {
	cmd := exec.Command("git", args...)
	cmd.Dir = dir
	output, err := cmd.CombinedOutput()
	if err != nil {
		return fmt.Errorf("%s: %w", bytes.TrimSpace(output), err)
	}
	return nil
}

func runGitOutput(dir string, args ...string) (string, error) {
	cmd := exec.Command("git", args...)
	cmd.Dir = dir
	output, err := cmd.CombinedOutput()
	if err != nil {
		return "", fmt.Errorf("%s: %w", bytes.TrimSpace(output), err)
	}
	return string(output), nil
}
```

```go
package plugin

import "context"

type ProviderRequest struct {
	Mode       Mode
	WorkDir    string
	Prompt     string
	Model      string
	SchemaFile string
	ExtraArgs  []string
}

type Provider interface {
	Run(ctx context.Context, request ProviderRequest) (string, error)
}

type commandRunner interface {
	Run(ctx context.Context, name string, args []string, dir string, stdin string) (string, error)
}
```

```go
package plugin

import (
	"context"
	"fmt"
)

type CodexProvider struct {
	bin    string
	runner commandRunner
}

func NewCodexProvider(bin string, runner commandRunner) Provider {
	return CodexProvider{bin: bin, runner: runner}
}

func (p CodexProvider) Run(ctx context.Context, request ProviderRequest) (string, error) {
	args := []string{"exec", "-C", request.WorkDir}
	if request.Model != "" {
		args = append(args, "--model", request.Model)
	}
	if request.Mode == ModePR {
		args = append(args, "--output-schema", request.SchemaFile, "-")
	} else {
		args = append(args, "-")
	}
	args = append(args, request.ExtraArgs...)
	output, err := p.runner.Run(ctx, p.bin, args, request.WorkDir, request.Prompt)
	if err != nil {
		return "", fmt.Errorf("codex exec %v: %w", args, err)
	}
	return output, nil
}
```

```go
package plugin

import (
	"context"
	"encoding/json"
	"fmt"
	"os"
)

type dependencies interface {
	LoadPRContext() (PRContext, error)
	BuildPRDiff(workDir, targetBranch string) (string, error)
	NewProvider(cfg Config) Provider
	WriteFile(path string, data []byte) error
}

func execute(cfg Config, deps dependencies) error {
	provider := deps.NewProvider(cfg)
	switch cfg.Mode {
	case ModePR:
		prCtx, err := deps.LoadPRContext()
		if err != nil {
			return err
		}
		diff, err := deps.BuildPRDiff(cfg.WorkDir, prCtx.TargetBranch)
		if err != nil {
			return err
		}
		prompt := BuildPRPrompt(PRPromptInput{
			RepoRoot: cfg.WorkDir,
			Diff:     diff,
		})
		raw, err := provider.Run(context.Background(), ProviderRequest{
			Mode:      cfg.Mode,
			WorkDir:   cfg.WorkDir,
			Prompt:    prompt,
			Model:     cfg.Model,
			ExtraArgs: cfg.ExtraArgs,
		})
		if err != nil {
			return err
		}
		review, err := ParsePRReview([]byte(raw))
		if err != nil {
			return err
		}
		body, err := json.MarshalIndent(review, "", "  ")
		if err != nil {
			return fmt.Errorf("marshal pr review: %w", err)
		}
		return deps.WriteFile(cfg.Output, body)
	case ModeFull:
		raw, err := provider.Run(context.Background(), ProviderRequest{
			Mode:      cfg.Mode,
			WorkDir:   cfg.WorkDir,
			Prompt:    BuildFullPrompt(cfg.WorkDir),
			Model:     cfg.Model,
			ExtraArgs: cfg.ExtraArgs,
		})
		if err != nil {
			return err
		}
		return WriteMarkdown(cfg.Output, raw)
	default:
		return fmt.Errorf("unsupported mode %q", cfg.Mode)
	}
}

func (p Plugin) Exec() error {
	cfg, err := ParseConfig()
	if err != nil {
		return err
	}
	return execute(cfg, defaultDependencies{})
}

type defaultDependencies struct{}

func (defaultDependencies) LoadPRContext() (PRContext, error) { return LoadPRContext() }
func (defaultDependencies) BuildPRDiff(workDir, targetBranch string) (string, error) {
	return BuildPRDiff(workDir, targetBranch)
}
func (defaultDependencies) NewProvider(cfg Config) Provider { return MustProvider(cfg) }
func (defaultDependencies) WriteFile(path string, data []byte) error {
	return os.WriteFile(path, data, 0o644)
}
```
